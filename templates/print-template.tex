% Make a slightly narrower truncation box, then center THAT box in the full header width.
% This avoids the "left-stuck" look of truncate's fixed-width box.
\newcommand{\CenteredTruncatedHeader}[1]{%
  \makebox[\headwidth][c]{%
    \truncate{0.86\headwidth}{#1}%
  }%
}

% Put header content always into the center field and switch by page parity
\fancyhead[C]{%
  \ifodd\value{page}%
    \CenteredTruncatedHeader{\MakeUppercase{\leftmark}}%
  \else%
    \CenteredTruncatedHeader{\MakeUppercase{\BookTitle}}%
  \fi%
}

% Part styling (smaller, elegant, not bold)
% ------------------------------------------------
\makeatletter
\renewcommand{\partname}{}

\def\@part[#1]#2{%
  \if@twocolumn
    \onecolumn
    \@tempswatrue
  \else
    \@tempswafalse
  \fi
  \ifnum \c@secnumdepth >-2\relax
    \refstepcounter{part}%
    \addcontentsline{toc}{part}{\thepart\hspace{1em}#1}%
  \else
    \addcontentsline{toc}{part}{#1}%
  \fi
  \markboth{}{}%
  \thispagestyle{plain}%
  \vspace*{25pt}
  \begin{center}
    {\normalfont\Large\mdseries \thepart\par}
    \vspace{12pt}
    {\normalfont\LARGE\mdseries #2\par}
  \end{center}
  \vspace{35pt}
  \newpage
  \if@tempswa \twocolumn \fi
}

\def\@spart#1{%
  \if@twocolumn
    \onecolumn
    \@tempswatrue
  \else
    \@tempswafalse
  \fi
  \markboth{}{}%
  \thispagestyle{plain}%
  \vspace*{25pt}
  \begin{center}
    {\normalfont\LARGE\mdseries #1\par}
  \end{center}
  \vspace{35pt}
  \newpage
  \if@tempswa \twocolumn \fi
}
\makeatother
